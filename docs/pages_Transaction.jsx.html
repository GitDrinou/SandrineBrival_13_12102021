import { useEffect } from "react"
import { useDispatch } from "react-redux"
import AccountCard from "../components/AccountCard"
import Header from "../components/common/Header"
import { fetchUser } from "../features/loginSlice"
import { TITLE_PAGE_TRANSACTION } from "../utils/constants"
import { fakeAccounts } from "../utils/fakeAccounts"
import TransactionsList from "../components/TransactionsList"

/**
 * COMPONENT Transaction
 * @param { * } match - matching params (transaction ID)
 * @returns the account's amount selected and transaction's list
 */
const Transaction = ({ match }) => {

    const dispatch = useDispatch()

    // Change the title of the web page
    // launch the fetchUser action only if session data is not null ( user is still connected )
    useEffect(() => {
        document.title = TITLE_PAGE_TRANSACTION
        if (sessionStorage.getItem("sKAB") !== null) dispatch(fetchUser(JSON.parse(sessionStorage.getItem("sKAB"))))
    }, [dispatch])

    const idTransaction = match.params.idTransaction    // get the Route param idTransaction
    
    const userAccount = fakeAccounts && fakeAccounts.find(account => account.id === parseInt(idTransaction))  // check the right account
    
    return (
        <div>
            <Header />
            <div className='mainUser bg-light'>
                <AccountCard 
                    type = {userAccount.type}
                    accountId = {userAccount.accountId}
                    amount = {userAccount.amount}
                    currency = {userAccount.currency}
                    status = {userAccount.status}
                    idTransaction = {userAccount.id}
                    isViewMode = {true}
                />
                <TransactionsList idTransaction={userAccount.id} />
            </div>            
        </div>
    )
}

export default Transaction
